    <script>
        let isPasswordVisible = false;
        let userSensitiveData = "";
        let encryptedData = "";
        let transmittedData = "";
        let decryptedData = "";

        // Toggle the visibility of the input field
        function toggleVisibility() {
            const inputField = document.getElementById("user-input");
            if (isPasswordVisible) {
                inputField.type = "password"; // Mask the input with *
            } else {
                inputField.type = "text"; // Show the actual input
            }
            isPasswordVisible = !isPasswordVisible;
        }

        // Clear the input field
        function clearText() {
            document.getElementById("user-input").value = "";
        }

        // Encrypt data
        function encryptData() {
            userSensitiveData = document.getElementById("user-input").value;
            if (userSensitiveData) {
                encryptedData = btoa(userSensitiveData); // Base64 encryption
                document.getElementById("encrypted-data").textContent = "Encrypted: " + encryptedData;
                resetOutput(["transmitted-data", "decrypted-data"]);
            } else {
                alert("Please enter sensitive data!");
            }
        }

        // Simulate data transmission with animation
        function transmitData() {
            if (encryptedData) {
                transmittedData = encryptedData; // Simulated transmission
                const animationElement = document.getElementById("animation");
                animationElement.style.display = "block"; // Show animation
                setTimeout(() => {
                    animationElement.style.display = "none"; // Hide animation after 2 seconds
                    document.getElementById("transmitted-data").textContent = "Transmitted: " + transmittedData;
                }, 2000);
                resetOutput(["decrypted-data"]);
            } else {
                alert("Please encrypt the data first!");
            }
        }

        // Decrypt data
        function decryptData() {
            if (transmittedData) {
                decryptedData = atob(transmittedData); // Base64 decryption
                document.getElementById("decrypted-data").textContent = "Decrypted: " + decryptedData;
            } else {
                alert("Please transmit the data first!");
            }
        }

        // Reset output for subsequent steps
        function resetOutput(ids) {
            ids.forEach(id => {
                document.getElementById(id).textContent = "";
            });
        }
    </script>
